// 處理掃描到的QR碼
function handleScannedCode(data) {
// 檢查是否為分段數據
const segmentMatch = data.match(/^(\d+)\/(\d+):(.*)/);
if (segmentMatch) {
// 解析分段數據
const currentSegment = parseInt(segmentMatch[1]);
const totalSegs = parseInt(segmentMatch[2]);
const segmentData = segmentMatch[3];
// 更新掃描UI
updateSegmentScanUI(currentSegment, totalSegs);
// 如果是第一段，初始化段數據
if (currentSegment === 1) {
totalSegments = totalSegs;
qrSegments = new Array(totalSegs);
scannedSegments = new Set();
}
// 如果總段數與之前掃描的不一致，顯示警告
if (totalSegments !== totalSegs) {
showWarning('檢測到不一致的QR碼，請從頭開始掃描');
resetQrSegments();
return;
}
// 存儲該段數據
qrSegments[currentSegment - 1] = segmentData;
scannedSegments.add(currentSegment);
// 檢查是否已掃描所有段
if (scannedSegments.size === totalSegments) {
// 合併所有段數據
const fullData = qrSegments.join('');
processQrCodeData(fullData);
}
} else if (data.startsWith('DISS_EXP:')) {
// 單一QR碼數據
processQrCodeData(data);
}
}

// 標準化器類 - 用於數據標準化和反標準化
class Normalizer {
constructor(data) {
this.xMean = 0;
this.xStd = 1;
this.yMean = 0;
this.yStd = 1;
if (data && data.length > 0) {
this.calculateParams(data);
}
}
// 計算標準化參數
calculateParams(data) {
// 計算X均值和標準差
let sumX = 0;
let sumY = 0;
const n = data.length;
data.forEach(point => {
sumX += point.x;
sumY += point.y;
});
this.xMean = sumX / n;
this.yMean = sumY / n;
// 計算標準差
let sumXSquaredDiff = 0;
let sumYSquaredDiff = 0;
data.forEach(point => {
sumXSquaredDiff += Math.pow(point.x - this.xMean, 2);
sumYSquaredDiff += Math.pow(point.y - this.yMean, 2);
});
this.xStd = Math.sqrt(sumXSquaredDiff / n) || 1; // 避免除以零
this.yStd = Math.sqrt(sumYSquaredDiff / n) || 1; // 避免除以零
}
// 標準化單個X值
normalizeX(x) {
return (x - this.xMean) / this.xStd;
}
// 標準化單個Y值
normalizeY(y) {
return (y - this.yMean) / this.yStd;
}
// 反標準化X值
denormalizeX(x) {
return x * this.xStd + this.xMean;
}
// 反標準化Y值
denormalizeY(y) {
return y * this.yStd + this.yMean;
}
// 標準化數據集
normalizeData(data) {
return data.map(point => ({
x: this.normalizeX(point.x),
y: this.normalizeY(point.y)
}));
}
// 將標準化後的模型參數轉換為原始尺度
denormalizeModel(model) {
// 在標準化空間中，模型方程為 y_norm = slope * x_norm + intercept
// 將其轉換回原始尺度 y = (y_norm * yStd + yMean)
// 代入標準化的公式 y_norm = (y - yMean) / yStd, x_norm = (x - xMean) / xStd
// (y - yMean) / yStd = slope * (x - xMean) / xStd + intercept
// y - yMean = slope * yStd / xStd * (x - xMean) + intercept * yStd
// y = slope * yStd / xStd * (x - xMean) + intercept * yStd + yMean
// y = slope * yStd / xStd * x - slope * yStd / xStd * xMean + intercept * yStd + yMean
// 原始空間的斜率 = 標準化空間的斜率 * (y標準差/x標準差)
const originalSlope = model.slope * (this.yStd / this.xStd);
// 原始空間的截距 = 標準化空間的截距 * y標準差 + y均值 - 原始空間的斜率 * x均值
const originalIntercept = model.intercept * this.yStd + this.yMean - originalSlope * this.xMean;
return {
slope: originalSlope,
intercept: originalIntercept,
r: model.r,
r2: model.r2,
controlVars: model.controlVars,
normalizer: this,
// 在原始尺度下的預測函數
predict: function(x) {
return this.slope * x + this.intercept;
}
};
}
}

// 解析CSV數據

dissolveTime: -1 // 溶解时间
};

columnIndices.stirringSpeed = index;
} else if (header.includes('溶質的大小因子')) {

// 批量梯度下降訓練 (Gradient Descent)
function gradientDescentTraining(data, model, learningRate, epochs) {
let currentEpoch = 0;
const n = data.length;
let previousLoss = Infinity;
let stableCount = 0; // 計數器，記錄損失值保持穩定的迭代次數
const convergenceThreshold = 6; // 至少需要連續6次迭代損失值保持穩定才認為收斂
// 极小的批次大小，确保UI更新频繁（每批只计算1-5个周期）
const microBatchSize = Math.max(1, Math.min(5, Math.floor(epochs / 200)));
// 总批次数
const totalBatches = Math.ceil(epochs / microBatchSize);
let currentBatch = 0;
// 微批次训练步骤 - 一次只处理极少量的数据
function processMicroBatch() {
if (currentEpoch >= epochs || trainingCancelled) {
if (!trainingCancelled) {
finishTrainingAndSaveModel();
} else {
finishTraining();
}
return;
}
const endEpoch = Math.min(currentEpoch + microBatchSize, epochs);
let latestLoss = 0;
// 处理少量周期
for (let epoch = currentEpoch; epoch < endEpoch; epoch++) {
// 计算梯度 (使用全部数据)
let gradientSlope = 0;
let gradientIntercept = 0;
for (let i = 0; i < n; i++) {
const point = data[i];
const prediction = model.predict(point.x);
const error = prediction - point.y;
gradientSlope += error * point.x;
gradientIntercept += error;
}
gradientSlope = (2 / n) * gradientSlope;
gradientIntercept = (2 / n) * gradientIntercept;
// 更新参数
model.slope = model.slope - learningRate * gradientSlope;
model.intercept = model.intercept - learningRate * gradientIntercept;
// 计算损失 (每个批次只计算一次损失，减少计算量)
if (epoch === endEpoch - 1) {
latestLoss = calculateMSE(data, model);
lossHistory.push({ epoch: epoch, loss: latestLoss });
// 检查损失值是否收敛
const currentLossFixed = latestLoss.toFixed(6);
const previousLossFixed = previousLoss.toFixed(6);
if (currentLossFixed === previousLossFixed) {
stableCount++;
if (stableCount >= convergenceThreshold) {
showSuccessMessage('模型已收斂！小數點後六位損失值不再變化。');
finishTrainingAndSaveModel();
return;
}
} else {
stableCount = 0;
}
previousLoss = latestLoss;
}
}
// 更新当前已处理的周期数和批次数
currentEpoch = endEpoch;
currentBatch++;
// 更新进度条和损失显示
const progress = (currentEpoch / epochs) * 100;
// 直接操作DOM更新进度，然后强制重排
progressBar.style.width = progress + '%';
progressText.textContent = Math.round(progress) + '%';
lossText.textContent = '損失: ' + latestLoss.toFixed(6);
// 强制浏览器重新计算布局 (强制重排)
void progressBar.offsetHeight;
// 每隔一定周期更新回归线
if (currentBatch % 20 === 0) {
updateChartWithModel(model);
}
// 使用较长的延迟来确保UI更新
setTimeout(processMicroBatch, 30);
}
// 完成训练并保存模型
function finishTrainingAndSaveModel() {
// 计算最终模型的评估指标
const { r, r2 } = calculateStats(data, model);
model.r = r;
model.r2 = r2;
// 如果数据已标准化，将模型转换回原始尺度
if (normalizer) {
model = normalizer.denormalizeModel(model);
}
// 存储最终模型
regressionModel = model;
updateModelDisplay(regressionModel);
finishTraining();
}
// 开始第一个微批次训练
setTimeout(processMicroBatch, 0);
}

function processMicroBatch() {

gradientSlope += error * point.x;

latestLoss = calculateMSE(data, model);

showSuccessMessage('模型已收斂！小數點後六位損失值不再變化。');
finishTrainingAndSaveModel();

}
} else {
// 如果損失值發生變化，重置計數器

}
// 更新上一次的損失值
previousLoss = latestLoss;
}
}
// 更新当前已处理的周期数和批次数
currentEpoch = endEpoch;
currentBatch++;
// 更新进度条和损失显示
const progress = (currentEpoch / epochs) * 100;

updateChartWithModel(model);
}

setTimeout(processMicroBatch, 30);
}
// 开始第一个微批次训练

}

// 隨機選擇批量索引
function getRandomBatch(dataSize, batchSize) {
const indices = Array.from({ length: dataSize }, (_, i) => i);
// 如果批量大小等於或大於數據大小，直接返回所有索引
if (batchSize >= dataSize) {
return indices;
}
// 隨機洗牌索引
for (let i = dataSize - 1; i > 0; i--) {
const j = Math.floor(Math.random() * (i + 1));
[indices[i], indices[j]] = [indices[j], indices[i]];
}
// 返回前 batchSize 個索引
return indices.slice(0, batchSize);
}
// 計算均方誤差 (MSE)
function calculateMSE(data, model) {
const n = data.length;
let sumSquaredError = 0;
for (let i = 0; i < n; i++) {
const point = data[i];
const prediction = model.predict(point.x);
const error = prediction - point.y;
sumSquaredError += error * error;
}
return sumSquaredError / n;
}
// 計算相關統計數據 (相關係數和決定係數)
function calculateStats(data, model) {
const n = data.length;
let sumY = 0;
let sumY2 = 0;
let sumPredicted = 0;
let sumPredicted2 = 0;
let sumYPredicted = 0;
// 計算真實值和預測值的統計數據
for (let i = 0; i < n; i++) {
const point = data[i];
const predicted = model.predict(point.x);
sumY += point.y;
sumY2 += point.y * point.y;
sumPredicted += predicted;
sumPredicted2 += predicted * predicted;
sumYPredicted += point.y * predicted;
}
const meanY = sumY / n;
// 計算相關係數
const numerator = n * sumYPredicted - sumY * sumPredicted;
const denominator = Math.sqrt((n * sumY2 - sumY * sumY) * (n * sumPredicted2 - sumPredicted * sumPredicted));
const r = numerator / denominator;
// 計算決定係數 (R²)
let ssr = 0; // 回歸平方和
let sst = 0; // 總平方和
for (let i = 0; i < n; i++) {
const point = data[i];
const predicted = model.predict(point.x);
ssr += Math.pow(predicted - meanY, 2);
sst += Math.pow(point.y - meanY, 2);
}
const r2 = ssr / sst;
return { r, r2 };
}
